import { Trigger } from '@inkeep/agents-sdk';
import { githubWebhookSecret } from '../../credentials/github-webhook-secret';

export const githubWebhookWithSignatureTrigger = new Trigger({
  id: 'webhook-optional',
  name: 'GitHub Webhook with Signature',
  enabled: true,
  messageTemplate: 'New push event from GitHub',
  signatureVerification: {
    algorithm: 'sha256',
    encoding: 'hex',
    signature: {
      source: 'header',
      key: 'x-hub-signature-256',
      prefix: 'sha256=',
    },
    signedComponents: [{
      source: 'header',
      key: 'x-custom-header',
      required: false,
    },
      {
        source: 'body',
        required: true,
      },
    ],
    componentJoin: {
      strategy: 'concatenate',
      separator: '',
    },
  },
  signingSecretCredentialReference: githubWebhookSecret,
});
